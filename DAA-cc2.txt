void floyd(int c[10][10],int n) 
{ 
int min(int,int); 
for(i=1;i<=n;i++) 
{ 
for(j=1;j<=n;j++) 
{ 
d[i][j]=c[i][j]; 
}

for(k=1;k<=n;k++) 
{ 
for(i=1;i<=n;i++) 
{

{ 
d[i][j]=min(d[i][j],d[i][k]+d[k][j]); 
} 
} 
} 
}
